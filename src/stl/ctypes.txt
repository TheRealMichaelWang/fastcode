static alpha_chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRTUVWXYZ"
static digit_chars = "0123456789"

proc is_alpha(c) {
	for char in alpha_chars {
		if char == c {
			return true
		}
	}
	return false
}

proc is_digit(c) {
	for char in digit_chars {
		if char == c {
			return true
		}
	}
	return false
}

proc is_alphanum(c) {
	if is_alpha(c) => true
	elif is_digit(c) => true
	else => false
}
