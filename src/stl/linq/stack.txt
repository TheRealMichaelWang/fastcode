import "stl/linq/_stdlinq.txt"

record stack {
	head
}

proc push(stack, value) {
	oldhead = stack.head
	newhead = new node
	newhead.value = value
	newhead.next = oldhead
	stack.head = ref newhead
}

proc pop(stack) {
	if stack.head == null {
		return null
	}
	value = stack.head.value
	stack.head = stack.head.next
	return value
}
