record node {
	next,
	value
}

proc empty(collection) => collection.head == null

proc count(collection) {
	current = collection.head
	count = 0
	while true {
		if current== null {
			break
		}
		count++
		if current.next == null {
			break
		}
		else {
			current = current.next
		}
	}
	return count
}
