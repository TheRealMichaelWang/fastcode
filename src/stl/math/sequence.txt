proc range(n) {
	range = array(n)
	i = n
	while i-- {
		range[i] = i
	}
	return range
}

proc arithmetic(base, step, count) {
	seq = array(count)
	seq[0] = base
	i = 1
	while i < count {
		seq[i] = seq[i - 1] + step
		i++
	}
	return seq 
}

proc geometric(base, ratio, count) {
	seq = array(count)
	seq[0] = base
	i = 1
	while i < count {
		seq[i] = ratio * seq[i - 1]
		i++
	}
	return seq
}
